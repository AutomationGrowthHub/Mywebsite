/* ── ANIMATED COUNTERS ── */
function animateCounter(el, target, prefix, suffix, duration) {
  let start = 0;
  const step = 16;
  const steps = duration / step;
  const inc = target / steps;
  const timer = setInterval(() => {
    start += inc;
    if (start >= target) { start = target; clearInterval(timer); }
    el.textContent = prefix + Math.round(start).toLocaleString() + suffix;
  }, step);
}
let countersDone = false;
function runCounters() {
  if (countersDone) return;
  countersDone = true;
  animateCounter(document.getElementById('cnt-price'), 164, '$', '', 1400);
  animateCounter(document.getElementById('cnt-save'), 1000, '$', 'K+', 1800);
  animateCounter(document.getElementById('cnt-tools'), 50, '', '+', 1200);
}
